from abc import ABC, abstractmethod

class CryptographicToken(ABC):
    def __init__(self, token_id):
        self.token_id = token_id

    @abstractmethod
    def generate_signature(self):
        pass

class RSAHardwareToken(CryptographicToken):
    def __init__(self, token_id, key_length):
        super().__init__(token_id)
        if key_length not in [2048, 4096]:
            raise ValueError("Secure keys must measure exactly 2048 or 4096 bits.")
        self.key_length = key_length

    def generate_signature(self):
        return f"RSA-SIG-[{self.token_id}]-VALIDATED-KEYLEN-{self.key_length}"
